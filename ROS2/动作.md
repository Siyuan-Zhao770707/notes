**服务器端的代码实现**
*learning_action -> learning_action ->action_move_server.py*
```python
import time
import rclpy
from rclpy.node import Node
from rclpy.action import ActionServer           # 动作服务端接口
from learning_interface.action import MoveCircle # 自定义动作接口

"""
动作服务端工作流程（与客户端对称，三个处理阶段）：
  阶段①：接收客户端目标 → execute_callback 被自动调用
  阶段②：循环发布执行进度 → publish_feedback()
  阶段③：执行完毕返回结果 → goal_handle.succeed() + return result
"""

class MoveCircleActionServer(Node):
    def __init__(self, name):
        super().__init__(name)
        # 创建动作服务端（节点，接口类型，动作名，执行回调）
        self._action_server = ActionServer(
            self,
            MoveCircle,           # 接口
            'move_circle',        # 动作名称（需与客户端一致）
            self.execute_callback # 收到目标后自动调用此回调
        )

    def execute_callback(self, goal_handle):
        """收到目标后执行，并在整个过程中处理反馈和结果"""
        self.get_logger().info('Moving circle...')
        feedback_msg = MoveCircle.Feedback()

        # ========== 阶段②：循环执行并发布进度 ==========
        for i in range(0, 360, 30):
            feedback_msg.state = i
            self.get_logger().info('Publishing feedback: %d' % feedback_msg.state)
            goal_handle.publish_feedback(feedback_msg)  # 发布反馈给客户端
            time.sleep(0.5)

        # ========== 阶段③：执行完毕，返回结果 ==========
        goal_handle.succeed()            # 标记目标成功完成
        result = MoveCircle.Result()
        result.finish = True
        return result                    # 返回结果 → 客户端收到 get_result 响应


def main(args=None):
    rclpy.init(args=args)
    node = MoveCircleActionServer("action_move_server")
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

**客户端代码的实现**
注释：客户端的代码多，是因为要处理服务器多个阶段的反馈，所以要设定多个回调函数
1. 当你向服务器发送目标的时候，___服务器向你反馈当前状态下服务器可不可以执行你的任务___
2. 如果服务器返回可以实现目标，___服务器向客户端反馈当前的机器人的动作执行状态___
3. 当目标执行完成后，___服务器向客户端发送当前目标执行成功的反馈___
三个阶段服务器发送三个反馈，客户端需要三个回调函数来处理这些反馈
*learning_action -> learning_action -> action_move_client.py*
```python
import rclpy
from rclpy.node import Node
from rclpy.action import ActionClient
from learning_interface.action import MoveCircle   # 自定义动作接口

"""
动作客户端工作流程（三个回调对应三个阶段）：
  阶段①：发送目标 → 服务端回复 接受/拒绝        → goal_response_callback
  阶段②：服务端接受后自动执行，期间持续发进度    → feedback_callback
  阶段③：执行完毕 → 服务端返回最终结果          → get_result_callback
"""

class MoveCircleActionClient(Node):
    def __init__(self, name):
        super().__init__(name)
        # 创建动作客户端（接口类型，节点，动作名）
        self._action_client = ActionClient(self, MoveCircle, 'move_circle')

    # ==================== 阶段①：发送目标 ====================
    def send_goal(self, enable):
        """向服务端发送动作目标"""
        goal_msg = MoveCircle.Goal()
        goal_msg.enable = enable

        self._action_client.wait_for_server()          # 等待服务端就绪

        # 异步发送目标，同时注册 feedback 回调（阶段②）
        self._send_goal_future = self._action_client.send_goal_async(
            goal_msg,
            feedback_callback=self.feedback_callback)  # ← 处理执行中进度

        # 注册回调，检查服务端是否接受目标（阶段①反馈）
        self._send_goal_future.add_done_callback(
            self.goal_response_callback)

    # ==================== 阶段① 反馈：接受/拒绝 ====================
    def goal_response_callback(self, future):
        """服务端回复：是否接受目标"""
        goal_handle = future.result()
        if not goal_handle.accepted:
            self.get_logger().info('Goal rejected :(')
            return

        self.get_logger().info('Goal accepted :)')     # 接受 → 自动开始执行

        # 异步等待最终结果（阶段③）
        self._get_result_future = goal_handle.get_result_async()
        self._get_result_future.add_done_callback(self.get_result_callback)

    # ==================== 阶段② 反馈：执行进度 ====================
    def feedback_callback(self, feedback_msg):
        """服务端周期性发布的执行进度"""
        feedback = feedback_msg.feedback
        self.get_logger().info('Received feedback: {%d}' % feedback.state)

    # ==================== 阶段③ 反馈：最终结果 ====================
    def get_result_callback(self, future):
        """服务端执行完毕，返回最终结果"""
        result = future.result().result
        self.get_logger().info('Result: {%d}' % result.finish)


def main(args=None):
    rclpy.init(args=args)
    node = MoveCircleActionClient("action_move_client")
    node.send_goal(True)
    rclpy.spin(node)       # 保持运行，等待回调触发
    node.destroy_node()
    rclpy.shutdown()
```
